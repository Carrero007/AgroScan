CREATE TABLE [dbo].[Login] (
    [Id]        INT           IDENTITY (1, 1) NOT NULL,
    [CPF]       VARCHAR (11)  NOT NULL,
    [Senha]     VARCHAR (50)  NOT NULL,
    [Latitude]  FLOAT (53)    NULL,
    [Longitude] FLOAT (53)    NULL,
    [Nome]      VARCHAR (100) DEFAULT ('') NOT NULL,
    [Whatsapp]  VARCHAR (20)  NULL,
    PRIMARY KEY CLUSTERED ([Id] ASC)
);

CREATE TABLE [dbo].[Diagnosticos] (
    [DiagnosticoId]       INT            IDENTITY (1, 1) NOT NULL,
    [PlantaId]            INT            NULL,
    [NomeDoenca]          NVARCHAR (200) NULL,
    [NomeCientifico]      NVARCHAR (200) NULL,
    [Confianca]           INT            NULL,
    [Tratamento]          NVARCHAR (MAX) NULL,
    [TratamentoEcologico] NVARCHAR (MAX) NULL,
    [RiscoPropagacao]     NVARCHAR (50)  NULL,
    [DataDiagnostico]     DATETIME       DEFAULT (getdate()) NULL,
    PRIMARY KEY CLUSTERED ([DiagnosticoId] ASC),
    FOREIGN KEY ([PlantaId]) REFERENCES [dbo].[Plantas] ([PlantaId])
);

CREATE TABLE [dbo].[Plantas] (
    [PlantaId]        INT            IDENTITY (1, 1) NOT NULL,
    [NomeCientifico]  NVARCHAR (255) NOT NULL,
    [NomePopular]     NVARCHAR (255) NULL,
    [OutrosNomes]     NVARCHAR (500) NULL,
    [Reino]           NVARCHAR (100) DEFAULT ('Plantae') NULL,
    [Filo]            NVARCHAR (100) NULL,
    [Classe]          NVARCHAR (100) NULL,
    [Ordem]           NVARCHAR (100) NULL,
    [Familia]         NVARCHAR (100) NULL,
    [Genero]          NVARCHAR (100) NULL,
    [Especie]         NVARCHAR (100) NULL,
    [TipoPlanta]      NVARCHAR (100) NULL,
    [CicloVida]       NVARCHAR (50)  NULL,
    [AlturaMin]       DECIMAL (5, 2) NULL,
    [AlturaMax]       DECIMAL (5, 2) NULL,
    [TaxaCrescimento] NVARCHAR (50)  NULL,
    [Clima]           NVARCHAR (100) NULL,
    [Luminosidade]    NVARCHAR (100) NULL,
    [Rega]            NVARCHAR (100) NULL,
    [TipoSolo]        NVARCHAR (255) NULL,
    [PHMin]           DECIMAL (3, 1) NULL,
    [PHMax]           DECIMAL (3, 1) NULL,
    [Origem]          NVARCHAR (255) NULL,
    [RegiaoNativa]    NVARCHAR (255) NULL,
    [EpocaPlantio]    NVARCHAR (100) NULL,
    [EpocaFloracao]   NVARCHAR (100) NULL,
    [EpocaColheita]   NVARCHAR (100) NULL,
    [EhMedicinal]     BIT            DEFAULT ((0)) NULL,
    [EhComestivel]    BIT            DEFAULT ((0)) NULL,
    [EhToxica]        BIT            DEFAULT ((0)) NULL,
    [Usos]            NVARCHAR (MAX) NULL,
    [Descricao]       NVARCHAR (MAX) NULL,
    [Observacoes]     NVARCHAR (MAX) NULL,
    [DataCriacao]     DATETIME       DEFAULT (getdate()) NULL,
    [DataAtualizacao] DATETIME       NULL,
    PRIMARY KEY CLUSTERED ([PlantaId] ASC)
);
